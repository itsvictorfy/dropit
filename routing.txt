	func setMaintenanceNow(c *gin.Context) {
		IsItScheduled = IsItScheduled.Now(c.Param("in"), c.Param("length"))
	
		processMaintenance("set", c.Param("tenant"))
	
		c.IndentedJSON(http.StatusOK, IsItScheduled)
	}
